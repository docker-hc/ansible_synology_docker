Step 7 — SSH Init Script roles/codeserver/files/init-ssh.sh
bash#!/bin/bash
# Install dropbear if not present
if ! command -v dropbear &> /dev/null; then
    apt-get update -qq
    apt-get install -y --no-install-recommends dropbear
fi

# Fix SSH directory ownership
chown -R root:root /root/.ssh
chmod 700 /root/.ssh
chmod 600 /root/.ssh/authorized_keys 2>/dev/null || true

# Start dropbear in background
mkdir -p /etc/dropbear
dropbear -p 2222 -E -R

Step 10 — roles/codeserver/tasks/main.yml
yaml- name: Ensure Code Server directories exist
  file:
    path: "{{ item.path }}"
    state: directory
    owner: "{{ item.owner }}"
    group: "{{ item.group }}"
    mode: "{{ item.mode }}"
  loop:
    - { path: "{{ docker_data_path }}/codeserver/config", owner: "{{ codeserver.puid }}", group: "{{ codeserver.pgid }}", mode: "0755" }
    - { path: "{{ docker_data_path }}/codeserver/workspace", owner: "{{ codeserver.puid }}", group: "{{ codeserver.pgid }}", mode: "0755" }
    - { path: "{{ docker_data_path }}/codeserver/.ssh", owner: "root", group: "root", mode: "0700" }
    - { path: "{{ docker_data_path }}/codeserver/custom-cont-init.d", owner: "root", group: "root", mode: "0755" }

- name: Copy SSH init script
  copy:
    src: init-ssh.sh
    dest: "{{ docker_data_path }}/codeserver/custom-cont-init.d/01-ssh.sh"
    owner: root
    group: root
    mode: "0755"

- name: Ensure authorized_keys exists with correct key
  copy:
    content: "{{ codeserver.ssh_public_key }}\n"
    dest: "{{ docker_data_path }}/codeserver/.ssh/authorized_keys"
    owner: root
    group: root
    mode: "0600"

- name: Ensure SSH host keys exist
  command: >
    ssh-keygen -t {{ item.type }}
    -f {{ docker_data_path }}/codeserver/.ssh/ssh_host_{{ item.type }}_key
    -N ""
  args:
    creates: "{{ docker_data_path }}/codeserver/.ssh/ssh_host_{{ item.type }}_key"
  loop:
    - { type: rsa }
    - { type: ed25519 }

- name: Set correct permissions on SSH host keys
  file:
    path: "{{ docker_data_path }}/codeserver/.ssh/ssh_host_{{ item }}_key"
    mode: "0600"
    owner: root
    group: root
  loop:
    - rsa
    - ed25519
  ignore_errors: "{{ ansible_check_mode }}"

- name: Deploy Code Server container
  community.docker.docker_container:
    name: "{{ codeserver.container_name }}"
    image: "{{ codeserver.image }}"
    state: started
    restart_policy: unless-stopped
    pull: false
    privileged: true
    networks:
      - name: "{{ docker_network_name }}"
    env:
      PASSWORD: "{{ codeserver.password }}"
      PUID: "{{ codeserver.puid }}"
      PGID: "{{ codeserver.pgid }}"
      TZ: "{{ codeserver.timezone }}"
    volumes:
      - "{{ docker_data_path }}/codeserver/config:/config"
      - "{{ docker_data_path }}/codeserver/workspace:/workspace"
      - "{{ docker_data_path }}/codeserver/.ssh:/root/.ssh"
      - "{{ docker_data_path }}/codeserver/custom-cont-init.d:/custom-cont-init.d"
    ports:
      - "{{ codeserver.port }}:8443"
      - "{{ codeserver.ssh_port }}:2222"
    devices:
      - "/dev/urandom:/dev/urandom:rwm"
      - "/dev/random:/dev/random:rwm"
    capabilities:
      - NET_ADMIN
      - SYS_PTRACE
    memory: "{{ codeserver.memory }}"
    memory_swap: "{{ codeserver.memory_swap }}"
    comparisons:
      volumes: strict
      env: strict
      networks: strict

Key Lessons Learned
✗ cpus:           not supported on Synology kernel — omit it
✗ openssh:        fails due to arc4random entropy — use dropbear
✗ DOCKER_MODS:    unreliable on Synology — use custom-cont-init.d
✗ custom-init:    must be mounted at /custom-cont-init.d not /config
✗ dropbear -F:    blocks code-server from starting — use background mode
✗ SUDO_PASSWORD:  causes arc4random crash — remove it
✗ latest tag:     skips major versions — upgrade one major at a time
✓ privileged:     required for /dev/urandom on Synology kernel
✓ dropbear:       works where openssh fails
✓ root owns .ssh: required for dropbear key auth
✓ config symlink: required for nextcloud v33 image structure
✓ vault:          all secrets encrypted, safe to commit
